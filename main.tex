def decode(text):
    retlist = []
    textfilt = ''
    text.encode("ascii","ignore")
    text = text.casefold()
    for x in text:
        if (ord(x) >= 97 and ord(x) <= 122):
            textfilt += x
    while textfilt:
        temp = textfilt[:1]
        retlist.append(ord(temp) - 97)
        textfilt = textfilt[1:]
    return retlist
